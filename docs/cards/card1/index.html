window.onload = function () {
  const imageContainer = document.getElementById('image-container');
  const images = [
    'https://img.freepik.com/free-photo/close-up-young-business-person-doing-internship_23-2149305365.jpg?t=st=1694589480~exp=1694590080~hmac=f3b5b001d27934e5de9413268e553f26c48816287a1392d1e5c87c5a5bf4dabb',
    'https://img.freepik.com/free-photo/high-angle-people-working-with-laptop_23-2149636255.jpg?t=st=1694589480~exp=1694590080~hmac=047e5a971632be6effdd67d63d2f69e3affd1deda6a4c841884a4511c058227b',
    'https://img.freepik.com/free-photo/african-american-man-working-open-air-man-suit-sunglasses-with-beard-using-laptop-sitting-terrace-rooftops-working-manager-technology-concept_74855-25538.jpg?t=st=1694589480~exp=1694590080~hmac=15ea10c37e4eaaf2936bc43affaf0b0eb32949b060ebd9742cf3d33473b919f8'
  ]; // Add your image file paths here
  let currentImageIndex = 0;

  function changeImage() {
    currentImageIndex = (currentImageIndex + 1) % images.length;
    const newImageSrc = images[currentImageIndex];
    const imageElement = imageContainer.querySelector('img');

    imageElement.style.opacity = 0;

    imageElement.src = newImageSrc;

    function fadeIn() {
      let opacity = 0;
      const fadeInInterval = setInterval(() => {
        opacity += 0.05;
        imageElement.style.opacity = opacity.toFixed(2);
        if (opacity >= 1) {
          clearInterval(fadeInInterval);
        }
      }, 50);
    }
    setTimeout(fadeIn, 100);
  }

  setTimeout(() => {
    setInterval(changeImage, 4000);
  }, 1000);
};
